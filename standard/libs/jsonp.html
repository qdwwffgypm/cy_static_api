查看源码： /static/js/libs/jsonp.js

PS：该对象在CY.jsonp 对象下；
1、JSONP请求数据方式；

引用方式

  • 引用方式一、 

    								// ====安装&引入====
								npm install cy_static
								import jsonp from "cy_static/jsonp";
								// ====使用====
								// 用法一
								jsonp("/test.do", {id : 11});
								//用法二
								jsonp("/test.do", 6000);
								//用法三
								jsonp("/test.do", {id : 11}, 6000);
								//用法四
								jsonp("/test.do", {id : 11}, function() {
									console.log('调用成功');
								});

    引用方式二、 

    								<script type="text/javascript" src="/static/js/libs/require2.3.6.js"></script>
								<script type="text/javascript">
									requirejs([ "/static/js/libs/jsonp.js" ], function(jsonp) {
										// 用法一
										jsonp("/test.do", {id : 11});
										//用法二
										jsonp("/test.do", 6000);
										//用法三
										jsonp("/test.do", {id : 11}, 6000);
										//用法四
										jsonp("/test.do", {id : 11}, function() {
											console.log('调用成功');
										});
									});
								</script>

    引用方式三、 

    									<script type="text/javascript" src="/static/js/libs/jsonp.js"></script>
									<script type="text/javascript">
										// 用法一
										CY.jsonp("/test.do", {id : 11});
										//用法二
										CY.jsonp("/test.do", 6000);
										//用法三
										CY.jsonp("/test.do", {id : 11}, 6000);
										//用法四
										CY.jsonp("/test.do", {id : 11}, function() {
											console.log('调用成功');
										});
									</script>

方法参数

  • jsonp(url : String, data : Object, timeout : Number, callback : Function) : Void

    JSONP请求数据方式；

    参数名 类型 默认值 描述 必需
    url String "" "请求URL地址" Y
    data Object {} 请求参数 N
    timeout Number 7000 请求超时时间(毫秒) N
    callback Function 请求成功回调函数 N
